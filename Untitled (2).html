# Computing metrics for each product in each one of the top 5 cities to give recommendations about sales improving in other cities
for city in top_5_cities.index:
    # Filter data for the current city
    city_data = df[df['CITY'] == city]
    
    # Compute metrics for each product line in the city
    product_metrics = city_data.groupby('PRODUCTLINE').agg({
        'SALES': 'sum',
        'QUANTITYORDERED': 'sum',
        'PRICEEACH': 'mean',
        'ORDERNUMBER': 'nunique'
    }).rename(columns={'SALES': 'Total Revenue', 'QUANTITYORDERED': 'Total Units Sold', 'PRICEEACH': 'Average Price per Unit', 'ORDERNUMBER': 'Total Number of Orders'})
    
    # Print city name
    print(f'\n\n {city}')
    
    # Print product metrics table
    print(product_metrics.to_string())
